    (function () {
      const $ = (id) => document.getElementById(id);

      // Inputs

      const inMode = $("in_mode");

      // Outputs
      const outPrimary = $("out_primary");
      const outSecondary = $("out_secondary");
      const outStatus = $("out_status");

      // Buttons
      const btnCalc = $("btn_calc");
      const btnReset = $("btn_reset");

      function toNum(v, fallback = 0) {
        const n = Number(v);
        return Number.isFinite(n) ? n : fallback;
      }

      function calculate() {
        // TODO: Replace with real math
        const x = toNum(inExample.value, 0);

        outPrimary.textContent = (x * 2).toFixed(2);
        outSecondary.textContent = (x * 3).toFixed(2);

        // Status: keep simple; normalized threshold colors later (polish phase)
        outStatus.textContent = "OK — calculation complete";
      }

      function reset() {
        inExample.value = 0;
        inMode.value = "a";
        outPrimary.textContent = "—";
        outSecondary.textContent = "—";
        outStatus.textContent = "Enter values and calculate.";
      }

      // If you want real-time outputs, call calculate() on input events and
      // REMOVE the Calculate button entirely. Otherwise keep button-driven.
      btnCalc?.addEventListener("click", calculate);
      btnReset?.addEventListener("click", reset);
