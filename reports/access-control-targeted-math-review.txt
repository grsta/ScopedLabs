1041:       interiorZones,

1307:   function calc() {
1308:     const cars = Math.max(0, Math.floor(n("cars")));
1309:     const banks = Math.max(1, Math.floor(n("banks")));
1310:     const floors = Math.max(0, Math.floor(n("floors")));
1311:     const dest = els.dest.value;
1312:     const placement = els.placement.value;
1313: 
1314:     if (cars <= 0) {
1315:       if (els.results) {
1316:         els.results.innerHTML = row("Error", "Enter Elevator Cars > 0");
1317:       }
1318: 
1319:       destroyChart();
1320:       lastMetrics = null;
1321:       currentReport = null;
1322:       updateExportControls();
1323:       return;
1324:     }
1325: 
1326:     let carReaders = 0;
1327:     let lobbyReaders = 0;
1328: 
1329:     if (placement === "car") {
1330:       carReaders = cars;
1331:     } else if (placement === "lobby") {
1332:       lobbyReaders = banks * getPerBankReaders(floors);
1333:     } else {
1334:       carReaders = cars;
1335:       lobbyReaders = banks * getPerBankReaders(floors);
1336:     }
1337: 
1338:     let dcsAdd = 0;
1339: 
1340:     if (dest === "yes") {
1341:       if (placement === "lobby" || placement === "both") {
1342:         dcsAdd = banks;
1343:       } else {
1344:         dcsAdd = Math.max(1, Math.round(banks * 0.5));
1345:       }
1346:     }
1347: 
1348:     const totalReaders = carReaders + lobbyReaders + dcsAdd;
1349: 
1350:     const complexityIndex = Math.round(
1351:       totalReaders +
1352:       floors * 1.5 +
1353:       banks * 4 +
1354:       (dest === "yes" ? 12 : 0) +
1355:       (placement === "both" ? 10 : placement === "lobby" ? 4 : 0)
1356:     );
1357: 
1358:     const status = getStatus(complexityIndex);
1359:     const guidance = getGuidance(status, placement, dest);
1360:     const insight = getInsight(status, dest, totalReaders);
1361: 
1362:     els.results.innerHTML = [
1363:       row("Cars", cars),
1364:       row("Banks", banks),
1365:       row("Secured Floors", floors),
1366:       row("Destination Control", dest === "yes" ? "YES" : "NO"),
1367:       row("Placement", placement.toUpperCase()),
1368:       row("Car Readers (est.)", carReaders),
1369:       row("Lobby Readers (est.)", lobbyReaders),
1370:       row("DCS Adders (est.)", dcsAdd),
1371:       row("Estimated Total Readers", totalReaders),
1372:       row("Planning Complexity", complexityIndex),
1373:       row("System Status", status),
1374:       row("Design Guidance", guidance),
1375:       row("Engineering Insight", insight)
1376:     ].join("");
1377: 
1378:     lastMetrics = {
1379:       totalReaders,
1380:       carReaders,
1381:       lobbyReaders,
1382:       dcsAdd,
1383:       complexityIndex
1384:     };
