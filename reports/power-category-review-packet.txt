 177:       input.effPct,
 178:       input.healthLossPct,
 179:       input.tempLossPct,
 180:       input.loadW,
 181:       input.spikeX
 182:     );
 183: 
 184:     const passWorst = worstMin >= input.floorMin && !overloadWorst;
 185:     const runtimeLossMin = baselineMin - worstMin;
 186:     const runtimeLossPct = baselineMin > 0 ? (runtimeLossMin / baselineMin) * 100 : 0;
 187: 
 188:     const worstLoad = input.loadW * input.spikeX;
 189:     const loadPressurePct = (worstLoad / input.upsMaxW) * 100;
 190: 
 191:     const overloadMetric = Math.min(loadPressurePct, 100);
 192:     const runtimeLossMetric = Math.min(Math.max(runtimeLossPct, 0), 100);
 193:     const floorGapMetric = worstMin >= input.floorMin
 194:       ? Math.max(0, 100 - worstMin)
 195:       : Math.min(100, 70 + (input.floorMin - worstMin) * 2);
 196: 
 197:     const statusPack = ScopedLabsAnalyzer.resolveStatus({
 198:       compositeScore: Math.max(overloadMetric, runtimeLossMetric, floorGapMetric),
 199:       metrics: [
 200:         {
 201:           label: "Load Pressure",
 202:           value: Number(overloadMetric.toFixed(1)),
 203:           displayValue: fmtPct(loadPressurePct, 1)
 204:         },
 205:         {
 206:           label: "Runtime Loss",
 207:           value: Number(runtimeLossMetric.toFixed(1)),
 208:           displayValue: fmtMin(runtimeLossMin, 1)
 209:         },
 210:         {
 211:           label: "Floor Tightness",
 212:           value: Number(floorGapMetric.toFixed(1)),
 213:           displayValue: `${fmtMin(worstMin, 1)} vs ${fmtMin(input.floorMin, 0)}`
 214:         }
 215:       ],
 216:       healthyMax: 60,
 217:       watchMax: 85
 218:     });
 219: 
 220:     let resultClass = "Worst-Case Acceptable";
 221:     if (overloadWorst) resultClass = "Worst-Case Overload";
 222:     else if (!passWorst) resultClass = "Worst-Case Runtime Fail";
 223:     else if (runtimeLossPct >= 40) resultClass = "Heavy Worst-Case Penalty";
 224: 
 225:     let interpretation = `Baseline runtime is about ${fmtMin(baselineMin, 1)}, while worst-case runtime drops to about ${fmtMin(worstMin, 1)} after applying ${fmtPct(input.healthLossPct, 0)} battery health loss, ${fmtPct(input.tempLossPct, 0)} temperature derate, ${fmtPct(100 - input.effPct, 0)} inverter loss, and a ${fmt(input.spikeX, 2)}x load spike. Worst-case instantaneous load becomes ${fmtWatts(worstLoad)}, compared with UPS capacity of ${fmtWatts(input.upsMaxW)}.`;
 226: 
 227:     if (overloadWorst) {
 228:       interpretation += ` The spike condition exceeds UPS output capability, so overload risk becomes the first failure mode before runtime math even matters.`;
 229:     } else if (!passWorst) {
 230:       interpretation += ` The design stays online, but worst-case runtime falls below the required floor target, so outage resilience is weaker than the planning goal.`;
 231:     } else if (runtimeLossPct >= 40) {
 232:       interpretation += ` The design still passes, but the bad-day penalty is large enough that aging, environment, and surge behavior are doing meaningful damage to resilience.`;
 233:     } else {
 234:       interpretation += ` Worst-case penalties are present but still reasonably controlled, so the runtime floor survives with usable planning margin.`;
 235:     }
 236: 
 237:     let dominantConstraint = "";
 238:     if (overloadWorst || loadPressurePct > 100) {
 239:       dominantConstraint = "Load pressure is the dominant limiter. The spike-adjusted demand is outrunning UPS output capability.";
 240:     } else if (!passWorst || worstMin < input.floorMin) {
 241:       dominantConstraint = "Runtime floor is the dominant limiter. The bad-day scenario erodes runtime below the minimum acceptable resilience target.";
 242:     } else if (runtimeLossPct >= 25) {
 243:       dominantConstraint = "Worst-case runtime loss is the dominant limiter. Derates and spike behavior are consuming too much of the original runtime margin.";
 244:     } else {
 245:       dominantConstraint = "The stress assumptions are reasonably balanced. Worst-case penalties remain within a practical planning range.";
 246:     }
 247: 
 248:     let guidance = "";
 249:     if (overloadWorst) {
 250:       guidance = "Reduce spike demand, increase UPS size, or separate startup surges before treating this system as resilient under worst-case conditions.";
 251:     } else if (!passWorst) {
 252:       guidance = "Increase usable battery energy, lower the floor target only if justified, or reduce worst-case load assumptions before depending on this design.";
 253:     } else if (runtimeLossPct >= 40) {
 254:       guidance = "The design passes, but with a heavy penalty. Recheck aging assumptions, room temperature, and startup surge behavior before finalizing it.";
 255:     } else {
 256:       guidance = "Worst-case margin is workable. Use this as the stress-tested planning baseline for final runtime and battery decisions.";
 257:     }
 258: 
 259:     return {
 260:       ok: true,
 261:       ...input,
 262:       overloadWorst,
 263:       baselineMin,
 264:       worstMin,
 265:       passWorst,
 266:       runtimeLossMin,
 267:       runtimeLossPct,
 268:       worstLoad,
 269:       loadPressurePct,
 270:       resultClass,
 271:       status: statusPack.status,
 272:       interpretation,
 273:       dominantConstraint,
 274:       guidance,
 275:       overloadMetric,
 276:       runtimeLossMetric,
 277:       floorGapMetric
 278:     };
 279:   }
